## Usage
```ts
const input = 'HelloWorld'

const result = lowercaseFirstLetter(input)

console.log(result)
// Output: "helloWorld"
```

## Type definition
```ts
/**
 * Converts the first letter of a string to lowercase while keeping the rest unchanged.
 *
 * @param str - The input string.
 * @returns The transformed string with the first letter in lowercase.
 */
export declare const lowercaseFirstLetter: (str: string) => string
```
